erDiagram
LogEntry {
    AutoField id
    DateTimeField action_time
    ForeignKey user
    ForeignKey content_type
    TextField object_id
    CharField object_repr
    PositiveSmallIntegerField action_flag
    TextField change_message
}
Permission {
    AutoField id
    CharField name
    ForeignKey content_type
    CharField codename
}
Group {
    AutoField id
    CharField name
    ManyToManyField permissions
}
ContentType {
    AutoField id
    CharField app_label
    CharField model
}
Session {
    CharField session_key
    TextField session_data
    DateTimeField expire_date
}
AuthToken {
    CharField digest
    CharField token_key
    ForeignKey user
    DateTimeField created
    DateTimeField expiry
}
Category {
    BigAutoField id
    CharField name
    SlugField slug
}
Product {
    BigAutoField id
    ForeignKey category
    CharField name
    SlugField slug
    TextField description
    DecimalField price
    FileField image
    FileField thumbnail
    DateTimeField date_added
    BooleanField is_active
    IntegerField stock
}
CustomUser {
    BigAutoField id
    CharField password
    DateTimeField last_login
    BooleanField is_superuser
    BooleanField is_staff
    BooleanField is_active
    DateTimeField date_joined
    CharField name
    CharField email
    BooleanField is_business
    CharField document
    IntegerField contact_number
    CharField address
    CharField description
    ManyToManyField groups
    ManyToManyField user_permissions
}
Order {
    ForeignKey user_id
    AutoField id
    TextField shipping_address
    CharField contact_number
    CharField status
    DecimalField amount
    FileField proof_of_payment
    DateTimeField created_at
    DateTimeField updated_at
}
OrderItem {
    AutoField id
    ForeignKey order
    ForeignKey product
    IntegerField quantity
    DecimalField price
}
